CREATE DATABASE escola01;
USE escola01;

CREATE TABLE aluno (
  num_matri int(11) NOT NULL,
  data_nasci date NOT NULL,
  rua varchar(100) DEFAULT NULL,
  bairro varchar(50) DEFAULT NULL,
  cep varchar(10) DEFAULT NULL,
  cidade varchar(50) DEFAULT NULL,
  genero char(1) DEFAULT NULL,
  nome varchar(100) NOT NULL
) ENGINE=INNODB DEFAULT CHARSET=UTF8; 

CREATE TABLE cursos (
  id_cursos int(11) NOT NULL,
  nome varchar(100) NOT NULL,
  turno varchar(20) NOT NULL,
  area varchar(50) NOT NULL
) ENGINE=INNODB DEFAULT CHARSET=UTF8; 


CREATE TABLE dependentes (
  nome varchar(100) NOT NULL,
  grau_parentesco varchar(50) DEFAULT NULL,
  cod_depe int(11) NOT NULL,
  data_nasc date DEFAULT NULL
) ENGINE=INNODB DEFAULT CHARSET=UTF8; 


CREATE TABLE disciplina (
  cod_disc int(11) NOT NULL,
  descricao varchar(200) NOT NULL,
  codigo_sala int(11) NOT NULL,
  FOREIGN KEY (codigo_sala) REFERENCES sala (cod_sala)
) ENGINE=INNODB DEFAULT CHARSET=UTF8;

CREATE TABLE docente (
  cod_doce int(11) NOT NULL,
  endereco varchar(200) NOT NULL,
  telefone bigint(11) NOT NULL,
  nome varchar(100) NOT NULL,
  codigo_dependente int(11) DEFAULT NULL,
  FOREIGN KEY (codigo_dependente) REFERENCES dependentes (cod_depe)
) ENGINE=INNODB DEFAULT CHARSET=UTF8; 


CREATE TABLE matricula (
  id int(11) NOT NULL,
  data_matri date NOT NULL,
  numero_matricula int(11) NOT NULL,
  FOREIGN KEY (numero_matricula) REFERENCES aluno (num_matri)
) ENGINE=INNODB DEFAULT CHARSET=UTF8; 


CREATE TABLE sala (
  cod_sala int(11) NOT NULL,
  nome varchar(50) NOT NULL,
  localizacao varchar(100) NOT NULL
) ENGINE=INNODB DEFAULT CHARSET=UTF8;

SELECT aluno.num_matri, aluno.nome, aluno.data_nasci, aluno.rua, aluno.bairro, aluno.cep, aluno.cidade, aluno.genero, matricula.data_matri

SELECT aluno.num_matri, aluno.nome, aluno.data_nasci, aluno.rua, aluno.bairro, aluno.cep, aluno.cidade, aluno.genero, cursos.nome, cursos.turno, cursos.area
FROM aluno
JOIN matricula ON matricula.numero_matricula = aluno.num_matri
JOIN cursos ON cursos.id_cursos = matricula.id;

SELECT dependentes.nome, nome_dependente, dependentes.grau_parentesco, dependentes.data_nasc, docente.nome, nome_docente, docente.endereco, docente.telefone

JOIN docente ON docente.codigo_dependente = dependentes.cod_depe;

SELECT disciplina.descricao, nome_disciplina, disciplina.codigo_sala, sala.nome, nome_sala, sala.localizacao
FROM disciplina
JOIN sala ON sala.cod_sala = disciplina.codigo_sala;

SELECT docente.nome, nome_docente, docente.endereco, docente.telefone, disciplina.descricao
