Métodos em Objetos

Tudo começa com objetos! Em JavaScript, objetos podem conter métodos - funções que pertencem ao objeto. Vamos criar nosso primeiro objeto com um método.

📄 scripts.js 
// 1 - métodos
const animal = {
  nome: "Bob",
  latir: function () {
    console.log("Au au");
  },
};

animal.latir();
⌨️ Console
$ node scripts.js

🎯 Visualização do Objeto

animal {Object}
nome: "Bob"
latir: function()
2

Aprofundando em Métodos: this

A palavra-chave this permite que um método acesse as propriedades do próprio objeto. Vamos criar getters e setters!

📄 scripts.js 
// 2 - aprofundando em métodos
const pessoa = {
  nome: "Roberto",

  getNome: function () {
    return this.nome;
  },

  setNome: function (novoNome) {
    this.nome = novoNome;
  },
};

console.log(pessoa.nome);
console.log(pessoa.getNome());

pessoa.setNome("Roberto Alves");
console.log(pessoa.getNome());
⌨️ Console
$ node scripts.js

🎯 Como funciona o this

O this dentro de getNome() refere-se ao objeto pessoa, permitindo acessar pessoa.nome.

3

Prototype - A Base de Tudo

Em JavaScript, tudo tem um prototype! Strings, números, arrays - todos herdam de prototypes. Vamos inspecionar o prototype de diferentes tipos.

📄 scripts.js 
// 3 - prototype
const text = "asd";

console.log(Object.getPrototypeOf(text));

const bool = true;

console.log(Object.getPrototypeOf(bool));

const arr = [];

console.log(arr.length);

console.log(Object.getPrototypeOf(arr) === Array.prototype);
⌨️ Console
$ node scripts.js

🎯 Cadeia de Prototypes

String.prototype
⬇
text ("asd")
4

Mais sobre Prototype

Podemos criar objetos herdando de outros objetos usando Object.create(). Isso estabelece uma cadeia de protótipos!

📄 scripts.js 
// 4 - mais sobre prototype
const myObject = {
  a: "b",
};

console.log(Object.getPrototypeOf(myObject));

console.log(Object.getPrototypeOf(myObject) === Object.prototype);

const mySecondObject = Object.create(myObject);

console.log(mySecondObject);

console.log(Object.getPrototypeOf(mySecondObject) === myObject);
⌨️ Console
$ node scripts.js

🎯 Cadeia Criada

Object.prototype
⬇
myObject {a: "b"}
⬇
mySecondObject {}

Classes Básicas com Prototype

Vamos simular classes usando prototypes! Criamos um objeto "classe" e instâncias que herdam dele.

📄 scripts.js 
// 5 - classes básicas
const cachorro = {
  raca: null,
};

const pastorAlemao = Object.create(cachorro);

pastorAlemao.raca = "Pastor Alemão";

const bulldog = Object.create(cachorro);

bulldog.raca = "Bulldog";

console.log(pastorAlemao.raca);
console.log(bulldog.raca);

$ node scripts.js

6

Funções como Classes

Antes do ES6, usávamos funções factory para criar objetos. Elas retornam novos objetos configurados!

📄 scripts.js 
// 6 - funcoes como classes
function criarCachorro(nome, raca) {
  const cachorro = Object.create({});
  cachorro.raca = raca;
  cachorro.nome = nome;
  return cachorro;
}

const bob = criarCachorro("Bob", "Vira lata");

console.log(bob);

const jack = criarCachorro("Jack", "Poodle");

console.log(jack);

console.log(Object.getPrototypeOf(jack));
⌨️ Console
$ node scripts.js

🎯 Factory Function

A função criarCachorro é uma "fábrica" que cria e retorna novos objetos configurados.

7

Funções Construtoras

Com new, funções se tornam construtoras! O this refere-se à nova instância sendo criada.

📄 scripts.js 
// 7 - funcoes construtoras
function Cachorro(nome, raca) {
  this.nome = nome;
  this.raca = raca;
}

const husky = new Cachorro("Ozzy", "Husky");

console.log(husky);
console.log(husky.nome);
console.log(husky.raca);
⌨️ Console
$ node scripts.js

🎯 O operador new faz:

  1. Cria um novo objeto vazio
  2. Define o prototype para Cachorro.prototype
  3. Executa a função com this = novo objeto
  4. Retorna o novo objeto
8

Métodos na Função Construtora

Adicionamos métodos ao prototype para que todas as instâncias compartilhem a mesma função, economizando memória!

📄 scripts.js 
// 8 - métodos na função construtora
function Cachorro(nome, raca) {
  this.nome = nome;
  this.raca = raca;
}

const husky = new Cachorro("Ozzy", "Husky");

Cachorro.prototype.uivar = function () {
  console.log("Auuu");
};

husky.uivar();

const pastor = new Cachorro("Rex", "Pastor");
pastor.uivar(); // Também funciona!
⌨️ Console
$ node scripts.js

🎯 Prototype Compartilhado

Cachorro.prototype
uivar: function()
husky
pastor

Classes ES6 - Açúcar Sintático

ES6 introduziu a palavra-chave class - é "açúcar sintático" sobre prototypes, mas deixa o código mais limpo!

📄 scripts.js 
// 9 - classes es6
class CachorroClasse {
  constructor(nome, raca) {
    this.nome = nome;
    this.raca = raca;
  }
}

const jeff = new CachorroClasse("Jeff", "Labrador");

console.log(jeff);

console.log(Object.getPrototypeOf(jeff));
⌨️ Console
$ node scripts.js

🎯 Sintaxe Mais Limpa

Antigo (ES5)
function Cachorro(n, r) {
  this.nome = n;
}
Novo (ES6)
class Cachorro {
  constructor(n) {
    this.nome = n;
  }
}
10

Mais sobre Classes

Métodos são definidos automaticamente no prototype. Propriedades estáticas precisam ser adicionadas ao prototype manualmente.

📄 scripts.js 
// 10 - mais sobre classes
class Caminhao {
  constructor(eixos, cor) {
    this.eixos = eixos;
    this.cor = cor;
  }

  descreverCaminhao() {
    console.log(
      `Este caminhão tem ${this.eixos} eixos e é da cor ${this.cor}.`
    );
  }
}

const scania = new Caminhao(6, "Vermelha");

scania.descreverCaminhao();

Caminhao.motor = 4.0;

const c2 = new Caminhao(4, "Preta");

console.log("Motor:", c2.motor); // undefined

Caminhao.prototype.motor = 4.0;

const c3 = new Caminhao(6, "Azul");

console.log("Motor:", c3.motor); // 4.0
⌨️ Console
$ node scripts.js

🎯 Diferença: Classe vs Prototype

Caminhao.motor = propriedade estática da classe (não herdada)

Caminhao.prototype.motor = herdada por todas as instâncias

11

Override por Prototype

Podemos definir valores padrão no prototype que serão usados se a instância não tiver o valor próprio.

📄 scripts.js 
// 11 - override por prototype
class Humano {
  constructor(nome, idade) {
    this.nome = nome;
    this.idade = idade;
  }
}

const roberto = new Humano("Roberto", 60);

console.log(roberto);
console.log("Idade:", roberto.idade);

Humano.prototype.idade = "Não definida";

const joao = new Humano("João");
console.log("João idade:", joao.idade);

console.log("Roberto idade:", roberto.idade);
console.log("Prototype idade:", Humano.prototype.idade);
⌨️ Console
$ node scripts.js

🎯 Ordem de Busca

1. Propriedade da instância → 2. Prototype → 3. Prototype do prototype...

12

Symbol - Propriedades Únicas

Symbol cria identificadores únicos. Útil para propriedades "privadas" ou metadados que não devem ser acessados diretamente.

📄 scripts.js 
// 12 - symbol
class Aviao {
  constructor(marca, turbinas) {
    this.marca = marca;
    this.turbinas = turbinas;
  }
}

const asas = Symbol();
const prefixo = Symbol();

Aviao.prototype[asas] = 2;
Aviao.prototype[prefixo] = "PR-";

const boeing = new Aviao("Boeing", 10);

console.log(boeing);
console.log("Asas:", boeing[asas]);
console.log("Prefixo:", boeing[prefixo]);
console.log("Prototype asas:", Aviao.prototype[asas]);
⌨️ Console
$ node scripts.js

🎯 Symbols são únicos

Cada Symbol é diferente, mesmo com a mesma descrição. Não aparecem em for...in ou Object.keys().

13

Getter e Setter

get e set permitem criar propriedades computadas. Acessamos como propriedades, mas executam funções!

📄 scripts.js 
// 13 - getter e setter
class Post {
  constructor(titulo, descricao, tags) {
    this.titulo = titulo;
    this.descricao = descricao;
    this.tags = tags;
  }

  get exibirTitulo() {
    return `Você está lendo: ${this.titulo}`;
  }

  set adicionarTags(tags) {
    const tagsArray = tags.split(", ");
    this.tags = tagsArray;
  }
}

const myPost = new Post("Algum post", "É um post sobre programação");

console.log(myPost.exibirTitulo);

myPost.adicionarTags = "programacao, javascript, js";

console.log(myPost.tags);
⌨️ Console
$ node scripts.js

🎯 Sintaxe de propriedade

Getter: parece propriedade

post.exibirTitulo

Setter: atribuição

post.tags = "a, b"
14

Herança com extends e super

extends cria uma subclasse. super() chama o construtor da classe pai. É a base da reutilização de código!

📄 scripts.js 
// 14 - Herança
class Mamifero {
  constructor(patas) {
    this.patas = patas;
  }
}

class Lobo extends Mamifero {
  constructor(patas, nome) {
    super(patas);
    this.nome = nome;
  }

  uivar() {
    console.log("Auuuu!");
  }
}

const shark = new Lobo(4, "Shark");

console.log(shark);
console.log("Patas:", shark.patas);
console.log("Nome:", shark.nome);
shark.uivar();
⌨️ Console
$ node scripts.js

🎯 Diagrama de Herança

Mamifero
patas
⬇ extends
Lobo
patas (herdado) + nome
⬇ new
shark
{ patas: 4, nome: "Shark" }

Instanceof - Verificando Tipos

instanceof verifica se um objeto é instância de uma classe, considerando toda a cadeia de herança!

📄 scripts.js 
// 15 - instanceof
class Mamifero {
  constructor(patas) {
    this.patas = patas;
  }
}

class Lobo extends Mamifero {
  constructor(patas, nome) {
    super(patas);
    this.nome = nome;
  }
}

const shark = new Lobo(4, "Shark");

console.log("shark instanceof Lobo:", shark instanceof Lobo);

console.log("Lobo instanceof Mamifero:", Lobo instanceof Mamifero);

console.log("new Lobo() instanceof Mamifero:", 
  new Lobo(4, "teste") instanceof Mamifero);

class Cachorro {}
class Post {}

console.log("new Post() instanceof Cachorro:", 
  new Post("a", "b") instanceof Cachorro);
⌨️ Console
$ node scripts.js
